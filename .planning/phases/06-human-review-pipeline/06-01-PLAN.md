    **2. Create `steps-designer.service.ts`:**
    ```typescript
    @Injectable()
    export class StepsDesignerService {
      constructor(
        @InjectRepository(Capability) private capabilityRepo,
        @InjectRepository(CapabilityVersion) private capVersionRepo,
        @InjectRepository(StepDefinition) private stepDefRepo,
        @InjectRepository(StepSkillBinding) private bindingRepo,
        @InjectRepository(StepTransitionRule) private stepTransitionRuleRepo,
        private dataSource: DataSource,
      ) {}

      async listCapabilities(): Promise<Capability[]> {
        return this.capabilityRepo.find({
          relations: ['versions', 'versions.stepDefinitions'],
          order: { name: 'ASC' },
        });
        // Note: ensure stepDefinitions ordered by stepOrder within the relation
        // Use QueryBuilder if .find() relation ordering is insufficient
      }

      async getCapabilityVersion(verId: string): Promise<CapabilityVersion> {
        return this.capVersionRepo.findOneOrFail({
          where: { id: verId },
          relations: ['stepDefinitions', 'stepDefinitions.bindings'],
          order: { stepDefinitions: { stepOrder: 'ASC' } },
        });
      }

      async updateSteps(verId: string, steps: StepItemDto[]): Promise<StepDefinition[]> {
        // Atomic transaction to prevent duplicate stepOrder values
        return this.dataSource.transaction(async (manager) => {
          const stepDefRepo = manager.getRepository(StepDefinition);

          // Assign stepOrder sequentially from the input array order (1-based)
          const savedSteps: StepDefinition[] = [];
          for (let i = 0; i < steps.length; i++) {
            const dto = steps[i];
            // Upsert by key within this capability version
            let existing = await stepDefRepo.findOne({
              where: { capabilityVersionId: verId, key: dto.key },
            });
            if (existing) {
              existing.name = dto.name;
              existing.stepOrder = i + 1;
              existing.isHumanRequired = dto.isHumanRequired;
              existing.isActive = dto.isActive ?? true;
              existing.skillKey = dto.skillKey ?? null;
              existing.llmModel = dto.llmModel ?? null;
              existing = await stepDefRepo.save(existing);
            } else {
              existing = await stepDefRepo.save(
                stepDefRepo.create({
                  capabilityVersionId: verId,
                  key: dto.key,
                  name: dto.name,
                  stepOrder: i + 1,
                  isHumanRequired: dto.isHumanRequired,
                  isActive: dto.isActive ?? true,
                  skillKey: dto.skillKey ?? null,
                  llmModel: dto.llmModel ?? null,
                }),
              );
            }
            savedSteps.push(existing);
          }
          return savedSteps;
        });
      }

      async getTransitions(stepId: string): Promise<StepTransitionRule[]> {
        return this.stepTransitionRuleRepo.find({
          where: { stepDefinitionId: stepId },
          order: { targetStepOrder: 'ASC' },
        });
      }

      async updateTransitions(
        stepId: string,
        transitions: TransitionConditionDto[],
      ): Promise<StepTransitionRule[]> {
        // Replace-all strategy: delete existing rules then insert new ones atomically
        return this.dataSource.transaction(async (manager) => {
          const repo = manager.getRepository(StepTransitionRule);
          await repo.delete({ stepDefinitionId: stepId });
          const newRules = transitions.map(t =>
            repo.create({
              stepDefinitionId: stepId,
              condition: t.condition,
              targetStepOrder: t.targetStepOrder,
            }),
          );
          return repo.save(newRules);
        });
      }
    }
    ```
