'use client'

import { useState, useEffect, useRef } from 'react'
import { useAuthStore } from '@/store/auth.store'
import ProtocolSearch from './ProtocolSearch'
import ComplaintHeader from './ComplaintHeader'
import LockBanner from './LockBanner'
import NoteForm from './NoteForm'
import ProgressBar from './ProgressBar'

interface Complaint {
  id: string
  protocolNumber: string
  protocoloPrestadora: string | null
  rawText: string
  status: string
  slaDeadline?: string | null
  riskLevel?: string | null
  tipologia?: string | null
}

interface LockState {
  held: boolean
  conflict: boolean
  lockedBy?: string
  lockedAt?: string
}

interface ExecutionState {
  id: string | null
  status: 'idle' | 'running' | 'paused_human' | 'completed' | 'failed'
  completedSteps: number
  totalSteps: number
}

export default function ProcessarClient() {
  const { user, token } = useAuthStore()
  const [complaint, setComplaint] = useState<Complaint | null>(null)
  const [lockState, setLockState] = useState<LockState>({ held: false, conflict: false })
  const [execution, setExecution] = useState<ExecutionState>({ id: null, status: 'idle', completedSteps: 0, totalSteps: 14 })
  const [saving, setSaving] = useState(false)
  const [starting, setStarting] = useState(false)
  const [noteValues, setNoteValues] = useState({ plano: '', valor: '', motivo: '', data: '', observacao: '' })
  const renewTimerRef = useRef<ReturnType<typeof setInterval> | null>(null)

  const authHeader = { Authorization: `Bearer ${token}` }

  // Lock renewal every 10 minutes
  useEffect(() => {
    if (!lockState.held || !complaint) return
    renewTimerRef.current = setInterval(async () => {
      await fetch(`/api/proxy/complaints/${complaint.id}/lock/renew`, { method: 'POST', headers: authHeader })
    }, 10 * 60 * 1000)
    return () => { if (renewTimerRef.current) clearInterval(renewTimerRef.current) }
  }, [lockState.held, complaint])

  const handleSearch = async (q: string) => {
    const res = await fetch(`/api/proxy/complaints/by-protocol?q=${encodeURIComponent(q)}`, { headers: authHeader })
    if (!res.ok) return
    const results: Complaint[] = await res.json()
    if (results.length === 0) return
    const found = results[0]
    setComplaint(found)
    setLockState({ held: false, conflict: false })
    setExecution({ id: null, status: 'idle', completedSteps: 0, totalSteps: 14 })
    // Acquire lock immediately
    await acquireLock(found.id)
  }

  const acquireLock = async (complaintId: string) => {
    const res = await fetch(`/api/proxy/complaints/${complaintId}/lock`, { method: 'POST', headers: authHeader })
    if (res.ok) {
      setLockState({ held: true, conflict: false })
    } else if (res.status === 409) {
      const body = await res.json()
      setLockState({ held: false, conflict: true, lockedBy: body.lockedBy, lockedAt: body.lockedAt })
    }
  }

  const handleForceRelease = async () => {
    if (!complaint) return
    await fetch(`/api/proxy/complaints/${complaint.id}/lock/force`, { method: 'DELETE', headers: authHeader })
    await acquireLock(complaint.id)
  }

  const hasNoteContent = Object.values(noteValues).some(v => v.trim().length > 0)

  const handleSaveNote = async () => {
    if (!complaint) return
    setSaving(true)
    const content = [
      noteValues.plano && `Plano: ${noteValues.plano}`,
      noteValues.valor && `Valor: ${noteValues.valor}`,
      noteValues.motivo && `Motivo: ${noteValues.motivo}`,
      noteValues.data && `Data ocorrência: ${noteValues.data}`,
      noteValues.observacao,
    ].filter(Boolean).join('\n')
    const parameters = {
      plano: noteValues.plano || null,
      valor: noteValues.valor || null,
      motivo: noteValues.motivo || null,
      dataOcorrencia: noteValues.data || null,
    }
    await fetch(`/api/proxy/complaints/${complaint.id}/notes`, {
      method: 'POST',
      headers: { ...authHeader, 'Content-Type': 'application/json' },
      body: JSON.stringify({ content, parameters }),
    })
    setSaving(false)
  }

  const handleStartProcessing = async () => {
    if (!complaint || !hasNoteContent) return
    setStarting(true)
    await handleSaveNote()
    const res = await fetch(`/api/proxy/complaints/${complaint.id}/executions/start`, {
      method: 'POST',
      headers: authHeader,
    })
    if (res.ok) {
      const exec = await res.json()
      setExecution({ id: exec.id, status: 'running', completedSteps: 0, totalSteps: 14 })
    }
    setStarting(false)
  }

  return (
    <div className="max-w-3xl mx-auto space-y-6 p-6">
      <h1 className="text-2xl font-semibold text-gray-900">Processar Reclamação</h1>

      <ProtocolSearch onSearch={handleSearch} />

      {complaint && (
        <>
          {lockState.conflict && (
            <LockBanner
              lockedBy={lockState.lockedBy!}
              lockedAt={lockState.lockedAt}
              canForce={user?.role === 'SUPERVISOR' || user?.role === 'ADMIN'}
              onForce={handleForceRelease}
              onRetry={() => acquireLock(complaint.id)}
            />
          )}

          <ComplaintHeader complaint={complaint} />

          {lockState.held && execution.status === 'idle' && (
            <>
              <NoteForm values={noteValues} onChange={setNoteValues} />
              <div className="flex gap-3">
                <button
                  onClick={handleSaveNote}
                  disabled={saving || !hasNoteContent}
                  className="px-4 py-2 text-sm border border-gray-300 rounded-md hover:bg-gray-50 disabled:opacity-50"
                >
                  {saving ? 'Salvando...' : 'Salvar Rascunho'}
                </button>
                <button
                  onClick={handleStartProcessing}
                  disabled={starting || !hasNoteContent}
                  className="px-6 py-2 text-sm bg-blue-600 text-white rounded-md hover:bg-blue-700 disabled:opacity-50 font-medium"
                >
                  {starting ? 'Iniciando...' : 'Iniciar Processamento'}
                </button>
              </div>
            </>
          )}

          {(execution.status === 'running' || execution.status === 'paused_human') && (
            <ProgressBar
              executionId={execution.id!}
              token={token!}
              onUpdate={(completed, total, status) =>
                setExecution(e => ({ ...e, completedSteps: completed, totalSteps: total, status }))
              }
            />
          )}
        </>
      )}
    </div>
  )
}

'use client'
import { useState } from 'react'

export default function ProtocolSearch({ onSearch }: { onSearch: (q: string) => void }) {
  const [q, setQ] = useState('')
  const [loading, setLoading] = useState(false)

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    if (!q.trim()) return
    setLoading(true)
    await onSearch(q.trim())
    setLoading(false)
  }

  return (
    <form onSubmit={handleSubmit} className="flex gap-2">
      <input
        type="text"
        value={q}
        onChange={e => setQ(e.target.value)}
        placeholder="Protocolo Anatel (15 dígitos) ou protocolo interno"
        className="flex-1 rounded-md border border-gray-300 px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-blue-500"
      />
      <button
        type="submit"
        disabled={loading || !q.trim()}
        className="px-4 py-2 bg-blue-600 text-white text-sm rounded-md hover:bg-blue-700 disabled:opacity-50"
      >
        {loading ? 'Buscando...' : 'Buscar'}
      </button>
    </form>
  )
}

  high: 'bg-red-100 text-red-800',
  critical: 'bg-red-200 text-red-900',
  medium: 'bg-yellow-100 text-yellow-800',
  low: 'bg-green-100 text-green-800',

**ProgressBar.tsx** — polls execution status every 3s:
```tsx
'use client'
import { useEffect, useRef } from 'react'

type ExecutionStatus = 'running' | 'paused_human' | 'completed' | 'failed' | 'idle'
const TERMINAL = new Set(['paused_human', 'completed', 'failed'])

interface Props {
  executionId: string
  token: string
  onUpdate: (completed: number, total: number, status: ExecutionStatus) => void
}

export default function ProgressBar({ executionId, token, onUpdate }: Props) {
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null)
  const totalRef = useRef(14)

  useEffect(() => {
    let stopped = false
    const poll = async () => {
      try {
        const res = await fetch(`/api/proxy/executions/${executionId}/steps`, {
          headers: { Authorization: `Bearer ${token}` },
        })
        if (!res.ok) return
        const steps: Array<{ status: string }> = await res.json()
        const completed = steps.filter(s => s.status === 'completed').length
        const total = Math.max(steps.length, totalRef.current)
        totalRef.current = total

        // Determine execution status from steps
        const hasRunning = steps.some(s => s.status === 'running')
        const hasPaused = steps.some(s => s.status === 'waiting_human')
        const hasFailed = steps.some(s => s.status === 'failed')
        const allDone = completed === total && total > 0

        const status: ExecutionStatus =
          hasPaused ? 'paused_human' :
          hasFailed ? 'failed' :
          allDone ? 'completed' :
          hasRunning ? 'running' : 'running'

        onUpdate(completed, total, status)

        if (!stopped && TERMINAL.has(status)) {
          if (intervalRef.current) clearInterval(intervalRef.current)
        }
      } catch { /* ignore */ }
    }

    poll()
    intervalRef.current = setInterval(poll, 3000)
    return () => {
      stopped = true
      if (intervalRef.current) clearInterval(intervalRef.current)
    }
  }, [executionId, token])

  return null // parent renders the UI using onUpdate state
}
```
