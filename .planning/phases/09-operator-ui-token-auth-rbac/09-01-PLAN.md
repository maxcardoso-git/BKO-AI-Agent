```typescript
import { Injectable, UnauthorizedException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import * as crypto from 'crypto';
import { AccessToken } from '../entities/access-token.entity';

@Injectable()
export class AccessTokenService {
  private readonly ttlDays: number;

  constructor(
    @InjectRepository(AccessToken)
    private readonly repo: Repository<AccessToken>,
  ) {
    this.ttlDays = parseInt(process.env.TOKEN_DEFAULT_TTL_DAYS ?? '30', 10);
  }

  async generateForUser(userId: string): Promise<AccessToken> {
    const token = crypto.randomBytes(32).toString('hex'); // 64 hex chars
    const expiresAt = new Date(Date.now() + this.ttlDays * 24 * 60 * 60 * 1000);
    return this.repo.save(this.repo.create({ userId, token, expiresAt, isActive: true }));
  }

  async validate(rawToken: string): Promise<AccessToken> {
    const record = await this.repo.findOne({
      where: { token: rawToken, isActive: true },
      relations: ['user'],
    });
    if (!record) throw new UnauthorizedException('Token inválido ou inativo');
    if (record.expiresAt < new Date()) {
      throw new UnauthorizedException('Token expirado, contate o administrador');
    }
    // Update lastUsedAt (fire-and-forget)
    this.repo.update(record.id, { lastUsedAt: new Date() }).catch(() => {});
    return record;
  }

  async revoke(id: string): Promise<void> {
    await this.repo.update(id, { isActive: false });
  }

  async findAll(): Promise<AccessToken[]> {
    return this.repo.find({
      relations: ['user'],
      order: { createdAt: 'DESC' },
    });
  }
}
```

```typescript
import { Injectable, ForbiddenException, HttpException, HttpStatus } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { TicketLock } from '../entities/ticket-lock.entity';

const LOCK_TTL_MS = 30 * 60 * 1000; // 30 minutes

@Injectable()
export class TicketLockService {
  constructor(
    @InjectRepository(TicketLock)
    private readonly lockRepo: Repository<TicketLock>,
  ) {}

  async acquire(complaintId: string, userId: string): Promise<TicketLock> {
    const existing = await this.lockRepo.findOne({
      where: { complaintId },
      relations: ['user'],
    });

    if (existing && existing.expiresAt > new Date() && existing.userId !== userId) {
      throw new HttpException(
        {
          message: `Ticket sendo tratado por ${existing.user?.name ?? 'outro usuário'}`,
          lockedBy: existing.user?.name ?? 'outro usuário',
          lockedAt: existing.lockedAt,
        },
        HttpStatus.CONFLICT,
      );
    }

    // Expired or same user — DELETE+INSERT (UNIQUE constraint on complaintId)
    await this.lockRepo.delete({ complaintId });
    return this.lockRepo.save(
      this.lockRepo.create({
        complaintId,
        userId,
        lockedAt: new Date(),
        expiresAt: new Date(Date.now() + LOCK_TTL_MS),
      }),
    );
  }

  async renew(complaintId: string, userId: string): Promise<TicketLock> {
    const lock = await this.lockRepo.findOne({ where: { complaintId, userId } });
    if (!lock) throw new ForbiddenException('Você não possui lock neste ticket');
    lock.expiresAt = new Date(Date.now() + LOCK_TTL_MS);
    return this.lockRepo.save(lock);
  }

  async release(complaintId: string, userId: string): Promise<void> {
    await this.lockRepo.delete({ complaintId, userId });
  }

  async forceRelease(complaintId: string): Promise<void> {
    await this.lockRepo.delete({ complaintId });
  }

  async findByComplaint(complaintId: string): Promise<TicketLock | null> {
    return this.lockRepo.findOne({
      where: { complaintId },
      relations: ['user'],
    });
  }
}
```

```typescript
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository, DataSource } from 'typeorm';
import { ComplaintUserNote } from '../entities/complaint-user-note.entity';
import { TimingEventService } from './timing-event.service';

@Injectable()
export class ComplaintUserNoteService {
  constructor(
    @InjectRepository(ComplaintUserNote)
    private readonly noteRepo: Repository<ComplaintUserNote>,
    private readonly timingEventService: TimingEventService,
    private readonly dataSource: DataSource,
  ) {}

  async create(
    complaintId: string,
    userId: string,
    content: string,
    parameters: Record<string, unknown> | null = null,
  ): Promise<ComplaintUserNote> {
    const current = await this.noteRepo.findOne({
      where: { complaintId, userId, isActive: true },
      order: { version: 'DESC' },
    });
    const nextVersion = current ? current.version + 1 : 1;

    let newNote!: ComplaintUserNote;
    await this.dataSource.transaction(async (em) => {
      if (current) {
        await em.update(ComplaintUserNote, { id: current.id }, { isActive: false });
      }
      newNote = await em.save(
        ComplaintUserNote,
        em.create(ComplaintUserNote, {
          complaintId,
          userId,
          content,
          parameters,
          version: nextVersion,
          isActive: true,
        }),
      );
    });

    // Emit note_saved timing event (AUDIT-TIMING-01)
    await this.timingEventService.emit('note_saved', complaintId, null, userId);

    return newNote;
  }

  async findByComplaint(complaintId: string): Promise<ComplaintUserNote[]> {
    return this.noteRepo.find({
      where: { complaintId },
      order: { version: 'DESC' },
      relations: ['user'],
    });
  }

  async findActiveByComplaint(complaintId: string): Promise<ComplaintUserNote | null> {
    return this.noteRepo.findOne({
      where: { complaintId, isActive: true },
      relations: ['user'],
    });
  }
}
```
