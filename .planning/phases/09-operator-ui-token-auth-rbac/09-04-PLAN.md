```tsx
'use client'

import { useState, useEffect } from 'react'
import { useAuthStore } from '@/store/auth.store'

interface TokenRecord {
  id: string
  token: string
  isActive: boolean
  expiresAt: string
  lastUsedAt: string | null
  createdAt: string
  user: { id: string; name: string; email: string }
}

interface User { id: string; name: string; email: string }

export default function TokensPage() {
  const { token } = useAuthStore()
  const [tokens, setTokens] = useState<TokenRecord[]>([])
  const [users, setUsers] = useState<User[]>([])
  const [showModal, setShowModal] = useState(false)
  const [selectedUserId, setSelectedUserId] = useState('')
  const [ttlDays, setTtlDays] = useState('')
  const [newToken, setNewToken] = useState<string | null>(null)
  const [copied, setCopied] = useState(false)
  const [loading, setLoading] = useState(false)

  const authHeader = { Authorization: `Bearer ${token}` }

  const fetchTokens = async () => {
    const res = await fetch('/api/proxy/admin/access-tokens', { headers: authHeader })
    if (res.ok) setTokens(await res.json())
  }

  const fetchUsers = async () => {
    const res = await fetch('/api/proxy/admin/users', { headers: authHeader })
    if (res.ok) setUsers(await res.json())
  }

  useEffect(() => { fetchTokens(); fetchUsers() }, [])

  const handleGenerate = async () => {
    if (!selectedUserId) return
    setLoading(true)
    const body: Record<string, unknown> = { userId: selectedUserId }
    if (ttlDays) body.ttlDays = parseInt(ttlDays)
    const res = await fetch('/api/proxy/admin/access-tokens/generate', {
      method: 'POST',
      headers: { ...authHeader, 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    })
    if (res.ok) {
      const record: TokenRecord = await res.json()
      setNewToken(record.token)
      await fetchTokens()
    }
    setLoading(false)
  }

  const handleRevoke = async (id: string) => {
    await fetch(`/api/proxy/admin/access-tokens/${id}/revoke`, { method: 'POST', headers: authHeader })
    await fetchTokens()
  }

  const handleCopy = () => {
    if (newToken) { navigator.clipboard.writeText(newToken); setCopied(true) }
  }

  const closeModal = () => { setShowModal(false); setNewToken(null); setCopied(false); setSelectedUserId(''); setTtlDays('') }

  return (
    <div className="p-6 space-y-4">
      <div className="flex items-center justify-between">
        <h1 className="text-2xl font-semibold text-gray-900">Tokens de Acesso</h1>
        <button onClick={() => setShowModal(true)} className="px-4 py-2 bg-blue-600 text-white text-sm rounded-md hover:bg-blue-700">
          Novo Token
        </button>
      </div>

      <div className="rounded-lg border border-gray-200 bg-white overflow-hidden">
        <table className="w-full text-sm">
          <thead className="bg-gray-50">
            <tr>
              <th className="px-4 py-3 text-left font-medium text-gray-600">Usuário</th>
              <th className="px-4 py-3 text-left font-medium text-gray-600">Token</th>
              <th className="px-4 py-3 text-left font-medium text-gray-600">Expira em</th>
              <th className="px-4 py-3 text-left font-medium text-gray-600">Último uso</th>
              <th className="px-4 py-3 text-left font-medium text-gray-600">Status</th>
              <th className="px-4 py-3" />
            </tr>
          </thead>
          <tbody className="divide-y divide-gray-100">
            {tokens.map(t => (
              <tr key={t.id}>
                <td className="px-4 py-3">
                  <p className="font-medium">{t.user.name}</p>
                  <p className="text-xs text-gray-500">{t.user.email}</p>
                </td>
                <td className="px-4 py-3 font-mono text-xs">{t.token.slice(0, 8)}...</td>
                <td className="px-4 py-3 text-xs">{new Date(t.expiresAt).toLocaleDateString('pt-BR')}</td>
                <td className="px-4 py-3 text-xs">{t.lastUsedAt ? new Date(t.lastUsedAt).toLocaleDateString('pt-BR') : 'Nunca'}</td>
                <td className="px-4 py-3">
                  <span className={`rounded-full px-2 py-0.5 text-xs ${t.isActive ? 'bg-green-100 text-green-800' : 'bg-gray-100 text-gray-600'}`}>
                    {t.isActive ? 'Ativo' : 'Revogado'}
                  </span>
                </td>
                <td className="px-4 py-3">
                  {t.isActive && (
                    <button onClick={() => handleRevoke(t.id)} className="text-xs text-red-600 hover:underline">
                      Revogar
                    </button>
                  )}
                </td>
              </tr>
            ))}
            {tokens.length === 0 && (
              <tr><td colSpan={6} className="px-4 py-8 text-center text-gray-400">Nenhum token cadastrado</td></tr>
            )}
          </tbody>
        </table>
      </div>

      {/* Novo Token Modal */}
      {showModal && (
        <div className="fixed inset-0 bg-black/40 flex items-center justify-center z-50">
          <div className="bg-white rounded-lg shadow-xl p-6 w-full max-w-md space-y-4">
            <h2 className="text-lg font-semibold">Novo Token de Acesso</h2>
            {!newToken ? (
              <>
                <div>
                  <label className="block text-sm text-gray-600 mb-1">Usuário</label>
                  <select value={selectedUserId} onChange={e => setSelectedUserId(e.target.value)}
                    className="w-full rounded border border-gray-300 px-2 py-1.5 text-sm">
                    <option value="">Selecione...</option>
                    {users.map(u => <option key={u.id} value={u.id}>{u.name} ({u.email})</option>)}
                  </select>
                </div>
                <div>
                  <label className="block text-sm text-gray-600 mb-1">Validade (dias, opcional)</label>
                  <input type="number" value={ttlDays} onChange={e => setTtlDays(e.target.value)}
                    placeholder="30 (padrão)" className="w-full rounded border border-gray-300 px-2 py-1.5 text-sm" />
                </div>
                <div className="flex gap-2 justify-end">
                  <button onClick={closeModal} className="px-4 py-2 text-sm border rounded">Cancelar</button>
                  <button onClick={handleGenerate} disabled={!selectedUserId || loading}
                    className="px-4 py-2 text-sm bg-blue-600 text-white rounded disabled:opacity-50">
                    {loading ? 'Gerando...' : 'Gerar Token'}
                  </button>
                </div>
              </>
            ) : (
              <>
                <p className="text-sm text-gray-600">Token gerado. Copie e envie ao operador — <strong>não será exibido novamente.</strong></p>
                <div className="flex gap-2 items-center">
                  <code className="flex-1 bg-gray-100 rounded px-3 py-2 text-xs font-mono break-all">{newToken}</code>
                  <button onClick={handleCopy} className="px-3 py-2 text-xs border rounded shrink-0">
                    {copied ? '✓ Copiado' : 'Copiar'}
                  </button>
                </div>
                <div className="flex justify-end">
                  <button onClick={closeModal} className="px-4 py-2 text-sm bg-blue-600 text-white rounded">Fechar</button>
                </div>
              </>
            )}
          </div>
        </div>
      )}
    </div>
  )
}
```

```tsx
'use client'

import { useState, useEffect } from 'react'
import { useAuthStore } from '@/store/auth.store'

interface Lock {
  id: string
  complaintId: string
  lockedAt: string
  expiresAt: string
  user: { name: string; email: string }
  complaint: { protocolNumber: string }
}

export default function LocksPage() {
  const { token } = useAuthStore()
  const [locks, setLocks] = useState<Lock[]>([])
  const [releasing, setReleasing] = useState<string | null>(null)

  const authHeader = { Authorization: `Bearer ${token}` }

  const fetchLocks = async () => {
    const res = await fetch('/api/proxy/admin/locks', { headers: authHeader })
    if (res.ok) setLocks(await res.json())
  }

  useEffect(() => { fetchLocks() }, [])

  const handleForceRelease = async (complaintId: string) => {
    setReleasing(complaintId)
    await fetch(`/api/proxy/complaints/${complaintId}/lock/force`, { method: 'DELETE', headers: authHeader })
    await fetchLocks()
    setReleasing(null)
  }

  const timeLeft = (expiresAt: string) => {
    const ms = new Date(expiresAt).getTime() - Date.now()
    if (ms <= 0) return 'Expirado'
    const mins = Math.floor(ms / 60000)
    return `${mins} min`
  }

  return (
    <div className="p-6 space-y-4">
      <div className="flex items-center justify-between">
        <h1 className="text-2xl font-semibold text-gray-900">Bloqueios Ativos</h1>
        <button onClick={fetchLocks} className="text-sm text-blue-600 hover:underline">Atualizar</button>
      </div>

      <div className="rounded-lg border border-gray-200 bg-white overflow-hidden">
        <table className="w-full text-sm">
          <thead className="bg-gray-50">
            <tr>
              <th className="px-4 py-3 text-left font-medium text-gray-600">Protocolo</th>
              <th className="px-4 py-3 text-left font-medium text-gray-600">Operador</th>
              <th className="px-4 py-3 text-left font-medium text-gray-600">Bloqueado em</th>
              <th className="px-4 py-3 text-left font-medium text-gray-600">Expira em</th>
              <th className="px-4 py-3" />
            </tr>
          </thead>
          <tbody className="divide-y divide-gray-100">
            {locks.map(l => (
              <tr key={l.id}>
                <td className="px-4 py-3 font-mono text-xs">{l.complaint?.protocolNumber ?? l.complaintId}</td>
                <td className="px-4 py-3">
                  <p className="font-medium">{l.user.name}</p>
                  <p className="text-xs text-gray-500">{l.user.email}</p>
                </td>
                <td className="px-4 py-3 text-xs">{new Date(l.lockedAt).toLocaleTimeString('pt-BR')}</td>
                <td className="px-4 py-3 text-xs">{timeLeft(l.expiresAt)}</td>
                <td className="px-4 py-3">
                  <button
                    onClick={() => handleForceRelease(l.complaintId)}
                    disabled={releasing === l.complaintId}
                    className="text-xs text-red-600 hover:underline disabled:opacity-50"
                  >
                    {releasing === l.complaintId ? 'Liberando...' : 'Forçar Liberação'}
                  </button>
                </td>
              </tr>
            ))}
            {locks.length === 0 && (
              <tr><td colSpan={5} className="px-4 py-8 text-center text-gray-400">Nenhum bloqueio ativo</td></tr>
            )}
          </tbody>
        </table>
      </div>
    </div>
  )
}
```
