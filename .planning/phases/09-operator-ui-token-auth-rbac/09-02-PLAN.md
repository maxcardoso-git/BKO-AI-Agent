Pattern (adapt to existing code style):
```typescript
import { NextRequest, NextResponse } from 'next/server'
// Import decrypt from session lib (jose-based, edge-compatible)
// decrypt takes a string — safe to use in Edge middleware

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl
  const sessionCookie = request.cookies.get('session')?.value

  // Decrypt session (jose — Edge compatible)
  const session = sessionCookie ? await decrypt(sessionCookie).catch(() => null) : null

  const isAuthenticated = !!session

  // Redirect unauthenticated from protected routes
  if (!isAuthenticated && (pathname.startsWith('/processar') || pathname.startsWith('/admin') || pathname.startsWith('/tickets'))) {
    return NextResponse.redirect(new URL('/login', request.url))
  }

  // Block OPERATOR from /admin routes
  if (isAuthenticated && session.role === 'OPERATOR' && pathname.startsWith('/admin')) {
    return NextResponse.redirect(new URL('/unauthorized', request.url))
  }

  // OPERATOR default: / → /processar
  if (isAuthenticated && session.role === 'OPERATOR' && pathname === '/') {
    return NextResponse.redirect(new URL('/processar', request.url))
  }

  // Redirect authenticated from /login to role-appropriate home
  if (isAuthenticated && pathname === '/login') {
    const home = session.role === 'OPERATOR' ? '/processar' : '/tickets'
    return NextResponse.redirect(new URL(home, request.url))
  }

  return NextResponse.next()
}

export const config = {
  matcher: ['/tickets/:path*', '/processar/:path*', '/admin/:path*', '/login', '/'],
}
```

```tsx
import { redirect } from 'next/navigation'
import { cookies } from 'next/headers'

interface Props {
  searchParams: { token?: string }
}

async function exchangeToken(token: string): Promise<{ access_token: string; user: { id: string; email: string; name: string; role: string } } | null> {
  try {
    const res = await fetch(`${process.env.BACKEND_URL}/api/auth/token-exchange`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ token }),
      cache: 'no-store',
    })
    if (!res.ok) return null
    return res.json()
  } catch {
    return null
  }
}

export default async function ProcessarPage({ searchParams }: Props) {
  const { token } = searchParams

  // TOKEN EXCHANGE FLOW (AUTH-TOKEN-04)
  if (token) {
    const result = await exchangeToken(token)

    // Invalid/expired token — show error (AUTH-TOKEN-05)
    if (!result) {
      return (
        <div className="flex min-h-screen items-center justify-center bg-gray-50">
          <div className="rounded-lg bg-white p-8 shadow-md text-center max-w-md">
            <div className="text-red-500 text-5xl mb-4">⚠️</div>
            <h1 className="text-xl font-semibold text-gray-900 mb-2">Token inválido ou expirado</h1>
            <p className="text-gray-600">Token expirado, contate o administrador</p>
          </div>
        </div>
      )
    }

    // Create session with the backend JWT
    // Use createSession from session lib — encrypt and set cookie
    // Then redirect to /processar (no token in URL)
    const { createSession } = await import('@/lib/session')
    await createSession(result.access_token, result.user)
    redirect('/processar')
  }

  // MAIN PROCESSAR SHELL (authenticated — token already validated by middleware)
  // Full UI implemented in 09-03; this is the placeholder shell
  return (
    <div className="p-6">
      <h1 className="text-2xl font-semibold text-gray-900 mb-6">Processar Reclamação</h1>
      <div className="text-gray-500">Carregando...</div>
    </div>
  )
}
```
