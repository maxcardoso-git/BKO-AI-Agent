**1.1 Create `validation-api.ts`** (at `BKOConsole/src/lib/validation-api.ts`):

```typescript
const BACKEND = process.env.NEXT_PUBLIC_API_URL ?? '';

export interface ValidateDecisionPayload {
  decision: 'approved' | 'corrected' | 'rejected';
  humanFinal?: string;
  correctionReason?: string;
  rejectionReason?: string;
  observations?: string;
  checklist?: Record<string, boolean>;
}

export async function submitValidationDecision(
  complaintId: string,
  token: string,
  payload: ValidateDecisionPayload,
): Promise<{ ok: boolean; status: number; data?: unknown; error?: string }> {
  try {
    const res = await fetch(`${BACKEND}/api/complaints/${complaintId}/validate`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${token}`,
      },
      body: JSON.stringify(payload),
    });
    if (!res.ok) {
      const error = await res.text();
      return { ok: false, status: res.status, error };
    }
    return { ok: true, status: res.status, data: await res.json() };
  } catch (err) {
    return { ok: false, status: 0, error: String(err) };
  }
}

export async function fetchValidationContext(
  protocolo: string,
  token: string,
): Promise<{
  complaint: any;
  execution: any;
  draftArtifact: any | null;
  complianceArtifact: any | null;
  templateArtifact: any | null;
  kbContextArtifact: any | null;
  operatorNote: any | null;
} | null> {
  // 1. Lookup complaint by protocol (matches ProcessarClient.tsx:158 pattern)
  const cRes = await fetch(`${BACKEND}/api/complaints/by-protocol?q=${encodeURIComponent(protocolo)}`, {
    headers: { Authorization: `Bearer ${token}` },
  });
  if (!cRes.ok) return null;
  const list = await cRes.json() as Array<{ id: string; protocolNumber: string }>;
  if (!list?.length) return null;
  const target = list[0];

  // Fetch full complaint detail (matches ProcessarClient.tsx:185)
  const detailRes = await fetch(`${BACKEND}/api/complaints/${target.id}`, {
    headers: { Authorization: `Bearer ${token}` },
  });
  const complaint = detailRes.ok ? await detailRes.json() : { id: target.id, protocolNumber: target.protocolNumber };

  // 2. Fetch latest execution
  const eRes = await fetch(`${BACKEND}/api/complaints/${complaint.id}/executions?latest=true`, {
    headers: { Authorization: `Bearer ${token}` },
  });
  const executions = eRes.ok ? await eRes.json() : [];
  const execution = Array.isArray(executions) ? executions[0] : executions;

  // 3. Fetch artifacts
  const aRes = await fetch(`${BACKEND}/api/complaints/${complaint.id}/artifacts`, {
    headers: { Authorization: `Bearer ${token}` },
  });
  const artifacts: any[] = aRes.ok ? await aRes.json() : [];
  const pick = (type: string) =>
    artifacts.filter(a => a.artifactType === type).sort((a, b) => +new Date(b.createdAt) - +new Date(a.createdAt))[0] ?? null;

  // 4. Fetch latest operator note (endpoint mirrors NoteForm pattern; if your project uses
  //    /notes?active=true instead, adjust to match — grep NoteForm.tsx for the existing call)
  const nRes = await fetch(`${BACKEND}/api/complaints/${complaint.id}/notes?latest=true`, {
    headers: { Authorization: `Bearer ${token}` },
  });
  const operatorNote = nRes.ok ? await nRes.json() : null;

  return {
    complaint,
    execution,
    draftArtifact: pick('draft_response'),
    complianceArtifact: pick('compliance_check'),
    templateArtifact: pick('iqi_template'),
    kbContextArtifact: pick('manual_context'),
    operatorNote,
  };
}
```

```tsx
import { useState } from 'react';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Textarea } from '@/components/ui/textarea';

interface Props {
  open: boolean;
  onClose: () => void;
  onConfirm: (reason: string) => Promise<void>;
  pending: boolean;
}

export default function RejectionModal({ open, onClose, onConfirm, pending }: Props) {
  const [reason, setReason] = useState('');
  const [touched, setTouched] = useState(false);
  const invalid = touched && reason.trim().length < 10;

  return (
    <Dialog open={open} onOpenChange={(v) => !v && onClose()}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Reprovar Rascunho</DialogTitle>
        </DialogHeader>
        <p className="text-sm text-muted-foreground">
          Descreva o motivo da reprovacao (min. 10 caracteres). A reclamacao sera cancelada e voce podera reprocessar com nota atualizada.
        </p>
        <Textarea
          value={reason}
          onChange={(e) => { setReason(e.target.value); setTouched(true); }}
          rows={5}
          placeholder="Ex.: Resposta cita fato incorreto sobre plano X"
          disabled={pending}
        />
        {invalid && <p className="text-xs text-red-600">Motivo deve ter ao menos 10 caracteres</p>}
        <DialogFooter>
          <Button variant="outline" onClick={onClose} disabled={pending}>Cancelar</Button>
          <Button
            variant="destructive"
            onClick={() => { setTouched(true); if (reason.trim().length >= 10) onConfirm(reason.trim()); }}
            disabled={pending || reason.trim().length < 10}
          >
            {pending ? 'Reprovando...' : 'Confirmar Reprovacao'}
          </Button>
        </DialogFooter>
      </DialogContent>
```

```tsx
'use client';

import { useEffect, useState, useCallback } from 'react';
import { useRouter } from 'next/navigation';
import { toast } from 'sonner';
import { useRequireAuth } from '@/hooks/use-require-auth';
import { useAuthStore } from '@/store/auth.store';
import { RichTextArea } from '@/components/ui/rich-text-area';
import { Button } from '@/components/ui/button';
import { Textarea } from '@/components/ui/textarea';
import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { ArrowLeft, Check, Edit3, X } from 'lucide-react';
import { submitValidationDecision, fetchValidationContext } from '@/lib/validation-api';
import RejectionModal from './RejectionModal';
import InjectedCorrectionsPanel from './InjectedCorrectionsPanel';

interface Props { protocolo: string }

export default function ValidarClient({ protocolo }: Props) {
  const { hasAccess, isLoading: authLoading } = useRequireAuth(['OPERATOR', 'SUPERVISOR', 'ADMIN']);
  const { token } = useAuthStore();
  const router = useRouter();

  const [ctx, setCtx] = useState<any>(null);
  const [loading, setLoading] = useState(true);
  const [draftText, setDraftText] = useState('');
  const [correctionReason, setCorrectionReason] = useState('');
  const [observations, setObservations] = useState('');
  const [pending, setPending] = useState<'approve' | 'correct' | 'reject' | null>(null);
  const [rejectOpen, setRejectOpen] = useState(false);
  const [correctionError, setCorrectionError] = useState('');

  const load = useCallback(async () => {
    if (!token) return;
    setLoading(true);
    const data = await fetchValidationContext(protocolo, token);
    if (!data) {
      toast.error('Reclamacao nao encontrada');
      router.push('/processar');
      return;
    }
    setCtx(data);
    setDraftText(data.draftArtifact?.content?.draftResponse ?? '');
    setLoading(false);
  }, [protocolo, token, router]);

  useEffect(() => { if (hasAccess && token) load(); }, [hasAccess, token, load]);

  const handleDecision = async (decision: 'approved' | 'corrected' | 'rejected', extra: any = {}) => {
    if (!ctx || !token) return;
    setPending(decision === 'approved' ? 'approve' : decision === 'corrected' ? 'correct' : 'reject');
    const payload: any = { decision, observations, ...extra };
    if (decision === 'approved' || decision === 'corrected') payload.humanFinal = draftText;
    if (decision === 'corrected') payload.correctionReason = correctionReason.trim();
    const res = await submitValidationDecision(ctx.complaint.id, token, payload);
    setPending(null);
    if (!res.ok) {
      toast.error(`Erro: ${res.error ?? res.status}`);
      return;
    }
    if (decision === 'approved') toast.success('Resposta aprovada');
    if (decision === 'corrected') toast.success('Correcao registrada');
    if (decision === 'rejected') {
      toast.success('Rascunho reprovado — voce pode atualizar a nota e reprocessar');
      router.push(`/processar?retry=${encodeURIComponent(ctx.complaint.protocolNumber)}`);
      return;
    }
    router.push('/processar');
  };

  if (authLoading || !hasAccess) return <div className="p-8">Carregando...</div>;
  if (loading || !ctx) return <div className="p-8">Carregando contexto...</div>;

  const injected = (ctx.draftArtifact?.content?.injectedCorrections ?? []) as any[];
  const conformanceScore = ctx.complianceArtifact?.content?.score ?? null;

  return (
    <div className="container mx-auto p-6 max-w-7xl">
      <button onClick={() => router.push('/processar')} className="flex items-center gap-2 text-sm text-muted-foreground hover:text-foreground mb-4">
        <ArrowLeft className="w-4 h-4" /> Voltar
      </button>

      <header className="flex justify-between items-start mb-6">
        <div>
          <h1 className="text-2xl font-bold">Validar Resposta</h1>
          <p className="text-sm text-muted-foreground">
            Protocolo: <strong>{ctx.complaint.protocolNumber}</strong> | Tipologia: {ctx.complaint.tipology?.name ?? ctx.complaint.tipologia ?? '—'}
          </p>
        </div>
        {conformanceScore != null && (
          <Badge variant={conformanceScore >= 80 ? 'default' : 'destructive'}>
            Conformidade: {conformanceScore}/100
          </Badge>
        )}
      </header>

      <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
        {/* LEFT: Editor + actions */}
        <div className="lg:col-span-2 space-y-4">
          <Card>
            <CardHeader><CardTitle>Rascunho da Resposta (editavel)</CardTitle></CardHeader>
            <CardContent>
              <RichTextArea value={draftText} onChange={setDraftText} rows={18} />
            </CardContent>
          </Card>

          <Card>
            <CardHeader><CardTitle>Razao da Correcao (obrigatoria se Corrigir)</CardTitle></CardHeader>
            <CardContent>
              <Textarea
                value={correctionReason}
                onChange={(e) => { setCorrectionReason(e.target.value); setCorrectionError(''); }}
                rows={3}
                placeholder="Ex.: Ajustei tom para mais empatico"
              />
              {correctionError && <p className="text-xs text-red-600 mt-1">{correctionError}</p>}
            </CardContent>
          </Card>

          <Card>
            <CardHeader><CardTitle>Observacoes (opcional)</CardTitle></CardHeader>
            <CardContent>
              <Textarea value={observations} onChange={(e) => setObservations(e.target.value)} rows={2} />
            </CardContent>
          </Card>

          <div className="flex gap-3">
            <Button onClick={() => handleDecision('approved')} disabled={pending !== null} className="flex-1">
              <Check className="w-4 h-4 mr-2" /> {pending === 'approve' ? 'Aprovando...' : 'Aprovar'}
            </Button>
            <Button
              variant="secondary"
              onClick={() => {
                if (correctionReason.trim().length < 5) { setCorrectionError('Informe a razao da correcao (min. 5 caracteres)'); return; }
                handleDecision('corrected');
              }}
              disabled={pending !== null}
              className="flex-1"
            >
              <Edit3 className="w-4 h-4 mr-2" /> {pending === 'correct' ? 'Salvando...' : 'Corrigir'}
            </Button>
            <Button variant="destructive" onClick={() => setRejectOpen(true)} disabled={pending !== null} className="flex-1">
              <X className="w-4 h-4 mr-2" /> Reprovar
            </Button>
          </div>
        </div>

        {/* RIGHT: Context panels */}
        <div className="space-y-4">
          <InjectedCorrectionsPanel corrections={injected} />

          {ctx.operatorNote && (
```

```typescript
export interface FeedbackRow {

export async function fetchAdminFeedback(
  token: string,
  filters: { tipologyId?: string; feedbackType?: string; limit?: number; offset?: number } = {}
): Promise<FeedbackListResult> {
  const params = new URLSearchParams();
  if (filters.tipologyId) params.set('tipologyId', filters.tipologyId);
  if (filters.feedbackType) params.set('feedbackType', filters.feedbackType);
  params.set('limit', String(filters.limit ?? 50));
  params.set('offset', String(filters.offset ?? 0));
  const res = await fetch(`${BACKEND}/api/admin/feedback?${params}`, {
    headers: { Authorization: `Bearer ${token}` },
  });
  if (!res.ok) throw new Error(`Failed to fetch feedback: ${res.status}`);
  return res.json();
}
```
